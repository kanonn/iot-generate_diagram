FROM eclipse-temurin:21-jre-jammy

# 1. 安装 fontconfig（Ubuntu 环境下的字体管理工具）
RUN apt-get update && apt-get install -y \
    fontconfig \
    && rm -rf /var/lib/apt/lists/*

# 2. 创建存放微软字体的目录
# Ubuntu 的标准路径通常是 /usr/share/fonts/truetype/
RUN mkdir -p /usr/share/fonts/truetype/windows

# 3. 将你本地的 msgothic.ttc 拷贝进去
# 请确保 msgothic.ttc 文件放在 Dockerfile 同级目录下
COPY msgothic.ttc /usr/share/fonts/truetype/windows/

# 4. 刷新字体缓存，让系统识别到 MS P Gothic
RUN fc-cache -f -v

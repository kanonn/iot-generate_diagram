# 假设你使用的是基于 Amazon Linux 2 的基础镜像
FROM public.ecr.aws/amazonlinux/amazonlinux:2

# 安装 fontconfig 工具，这是 Linux 管理字体的基础
RUN yum install -y fontconfig xorg-x11-font-utils

# 创建字体目录
RUN mkdir -p /usr/share/fonts/windows

# 将你本地的字体文件拷贝进去 (假设你把 msgothic.ttc 放在了 Dockerfile 同级目录下)
COPY msgothic.ttc /usr/share/fonts/windows/

# 刷新系统字体缓存
RUN fc-cache -f -v
